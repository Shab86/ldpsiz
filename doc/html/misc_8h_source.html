104 static inline int encodeDiploid(unsigned char *gtype, unsigned gtypeSize,
105  const char *str) {
106  register unsigned char *g = gtype;
107  register const char *s = str;
108  register unsigned oneBit, twoBit;
109 
110  /*
111  * The code below loops over the input string
112  */
113  while(*s != '\0' && g < gtype + gtypeSize) {
114  if(*s == 'h') {
115  *g = UNPHASED_HETEROZYGOTE;
116  } else {
117  twoBit = encode01(*s);
118  if(twoBit > 1)
119  eprintf("ERR@%s:%d: bad value in genotype: \"%c\". str=%s\n",
120  __FILE__, __LINE__, *s, str);
121 
122  /*
123  * In the final genotype in the string is incorrect,
124  * consisting of single character rather than two, then
125  * the line below will read the '\0' at the end of the
126  * string. This will generate an error but not a
127  * segmentation fault.
128  */
129  ++s;
130  if(*s == '\0')
131  eprintf("ERR@%s:%d: : malformed input: %s\n",
132  __FILE__, __LINE__, str);
133  oneBit = encode01(*s);
134  if(oneBit > 1)
135  eprintf("ERR@%s:%d: bad value in genotype: \"%c\". str=%s\n",
136  __FILE__, __LINE__, *s, str);
137 
138  *g = 2 * twoBit + oneBit; /* either 0, 1, 2, or 3 */
139  }
140  ++g;
141  ++s;
142  }
